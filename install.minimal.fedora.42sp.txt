install nerd fonts
	$ mkdir -p ~/Downloads
	$ cd ~/Downloads
	$ wget https://github.com/ryanoasis/nerd-fonts/releases/download/v3.4.0/InconsolataLGC.zip
	$ unzip InconsolataLGC.zip -d InconsolataLGC
	$ mkdir -p ~/.local/share/fonts
	$ cp InconsolataLGC/*.ttf ~/.local/share/fonts/
	$ fc-cache -fv
	$ cd ; rm -rf Downloads/*

autoinit startx
	$ vim ~/.zprofile
		+
		if [[ -z $DISPLAY && $(tty) == /dev/tty1 ]]; then
    		startx
		fi

configure terminator
	$ vim ~/.config/terminator/config
		+
		[global_config]
		  window_state = maximise
		[keybindings]
		[profiles]
		  [[default]]
		    font = Inconsolata LGC Nerd Font Propo Bold 12
		    use_system_font = False
		[layouts]
		  [[default]]
		    [[[window0]]]
		      type = Window
		      parent = ""
		    [[[child1]]]
		      type = Terminal
		      parent = window0
		[plugins]


configure openbox
	$ vim ~/.config/openbox/theme.xml
		+
		<openbox_config>
		  <theme>
		    <font place="MenuTitle">
		      <name>InconsolataLGC</name>
		      <size>12</size>
		      <weight>Normal</weight>
		      <slant>Normal</slant>
		    </font>
		    <font place="MenuItem">
		      <name>InconsolataLGC</name>
		      <size>12</size>
		      <weight>Normal</weight>
		      <slant>Normal</slant>
		    </font>
		  </theme>
		</openbox_config>
	$ cp /etc/xdg/openbox/rc.xml ~/.config/openbox/
	$ vim ~/.config/openbox/rc.xml
		+
		<keyboard>
		  <!-- Existing keybindings -->
		  <keybind key="C-A-T">
		    <action name="Execute">
		      <command>terminator</command>
		    </action>
		  </keybind>
		</keyboard>
	$ openbox --reconfigure
