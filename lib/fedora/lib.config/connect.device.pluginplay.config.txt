Install libs
  $ sudo dnf install simple-mtpfs gvfs-mtp kio_mtp libmtp libmtp*

Verify device
  $ dmesg | grep usb

Create rule
  $ sudo nano /etc/udev/rules.d/51-device.rules
    SUBSYSTEM=="usb", ATTR{idVendor}=="idvender.dmesg", ATTR{idProduct}=="idproduct.dmesg", MODE="0666", GROUP="plugdev"

Reset rules
  $ udevadm control --reload-rules

Create path
  $ mkdir -p /media/"device name"
  $ chmod a+rwx /media/"device name"

Config fuse
  $ sudo nano /etc/fuse.conf
    # mount_max = 1000
    user_allow_other

reconnect the device
