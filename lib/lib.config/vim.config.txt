function! OpenLexplore()
  if bufwinid(get(t:, 'netrw_lexbufnr')) >= 0
    return
  endif

  Lexplore
  vertical resize 25
  wincmd p
endfunction
augroup tab_lexplore
  autocmd!
  autocmd TabEnter,VimEnter * call OpenLexplore()
augroup END
